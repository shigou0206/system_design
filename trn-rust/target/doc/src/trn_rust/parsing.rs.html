12/// Parse a TRN string into a Trn object
13pub fn parse_trn(input: &str) -> TrnResult<Trn> {
14    // Simple regex-based parsing for better reliability
15    if !TRN_REGEX.is_match(input) {
16        return Err(TrnError::format(
17            "Invalid TRN format".to_string(),
18            Some(input.to_string()),
19        ));
20    }
21    
22    // Split TRN into main part and hash
23    let (main_part, hash) = if let Some(at_pos) = input.find('@') {
24        (&input[..at_pos], Some(&input[at_pos + 1..]))
25    } else {
26        (input, None)
27    };
28    
29    // Split main part by colons
30    let parts: Vec<&str> = main_part.split(':').collect();
31    
32    if parts.len() < 6 || parts[0] != "trn" {
33        return Err(TrnError::format(
34            "TRN must have at least 6 components and start with 'trn:'".to_string(),
35            Some(input.to_string()),
36        ));
37    }
38    
39    // Parse components based on the number of parts
40    let (platform, scope, resource_type, type_, subtype, instance_id, version, tag) = match parts.len() {
41        6 => {
42            // trn:platform:resource_type:type:instance_id:version
43            (parts[1], None, parts[2], parts[3], None, parts[4], parts[5], None)
44        }
45        7 => {
46            // Could be with scope or subtype - use heuristics
47            if is_scope_like(parts[2], parts[1]) {
48                // trn:platform:scope:resource_type:type:instance_id:version
49                (parts[1], Some(parts[2]), parts[3], parts[4], None, parts[5], parts[6], None)
50            } else {
51                // trn:platform:resource_type:type:subtype:instance_id:version
52                (parts[1], None, parts[2], parts[3], Some(parts[4]), parts[5], parts[6], None)
53            }
54        }
55        8 => {
56            // Could be scope+subtype or scope+tag
57            if is_scope_like(parts[2], parts[1]) {
58                // trn:platform:scope:resource_type:type:subtype:instance_id:version
59                (parts[1], Some(parts[2]), parts[3], parts[4], Some(parts[5]), parts[6], parts[7], None)
60            } else {
61                // trn:platform:resource_type:type:subtype:instance_id:version:tag
62                (parts[1], None, parts[2], parts[3], Some(parts[4]), parts[5], parts[6], Some(parts[7]))
63            }
64        }
65        9 => {
66            // trn:platform:scope:resource_type:type:subtype:instance_id:version:tag
67            (parts[1], Some(parts[2]), parts[3], parts[4], Some(parts[5]), parts[6], parts[7], Some(parts[8]))
68        }
69        _ => {
70            return Err(TrnError::format(
71                format!("Invalid number of TRN components: {}", parts.len()),
72                Some(input.to_string()),
73            ));
74        }
75    };
76    
77    // Create and validate the TRN
78    let trn = Trn::new_full(
79        platform.to_string(),
80        scope.map(String::from),
81        resource_type.to_string(),
82        type_.to_string(),
83        subtype.map(String::from),
84        instance_id.to_string(),
85        version.to_string(),
86        tag.map(String::from),
87        hash.map(String::from),
88    )?;
89    
90    Ok(trn)
91}

163/// Normalize TRN string
164pub fn normalize_trn(input: &str) -> TrnResult<String> {
165    let trn = parse_trn(input)?;
166    Ok(trn.to_string())
167}

169/// Parse with error recovery
170pub fn parse_trn_with_recovery(input: &str) -> TrnResult<Trn> {
171    // Try main parser first
172    match parse_trn(input) {
173        Ok(trn) => Ok(trn),
174        Err(_) => {
175            // Try with normalization
176            let normalized = input.to_lowercase();
177            parse_trn(&normalized)
178        }
179    }
180}

182/// Analyze parsing errors for better diagnostics
183fn analyze_parse_error(input: &str) -> TrnError {
184    if !input.starts_with("trn:") {
185        return TrnError::format("TRN must start with 'trn:'".to_string(), Some(input.to_string()));
186    }
187    
188    let parts: Vec<&str> = input.split(':').collect();
189    if parts.len() < 6 {
190        return TrnError::format(
191            format!("TRN requires at least 6 components, found {}", parts.len()),
192            Some(input.to_string()),
193        );
194    }
195    
196    TrnError::format("Unknown parsing error".to_string(), Some(input.to_string()))
197}

246/// Parse TRN from URL format
247pub fn parse_trn_from_url(url: &str) -> TrnResult<Trn> {
248    if url.starts_with("trn://") {
249        parse_trn_url(url)
250    } else {
251        parse_trn(url)
252    }
253}

255/// Parse TRN URL format
256fn parse_trn_url(url: &str) -> TrnResult<Trn> {
257    if !url.starts_with("trn://") {
258        return Err(TrnError::format(
259            "URL must start with trn://".to_string(),
260            Some(url.to_string()),
261        ));
262    }
263    
264    // Remove trn:// prefix
265    let path = &url[6..];
266    
267    // Split query parameters if any
268    let (path, query) = if let Some(q_pos) = path.find('?') {
269        (&path[..q_pos], Some(&path[q_pos + 1..]))
270    } else {
271        (path, None)
272    };
273    
274    // Parse hash from query parameters
275    let hash = if let Some(query_str) = query {
276        parse_query_params(query_str).get("hash").cloned()
277    } else {
278        None
279    };
280    
281    // Split path components
282    let path_parts: Vec<&str> = path.split('/').filter(|s| !s.is_empty()).collect();
283    
284    if path_parts.len() < 5 {
285        return Err(TrnError::format(
286            "TRN URL requires at least 5 path components".to_string(),
287            Some(url.to_string()),
288        ));
289    }
290    
291    // Decode URL components and build TRN string
292    let decoded_parts: Result<Vec<String>, _> = path_parts
293        .iter()
294        .map(|part| percent_decode_str(part).decode_utf8().map(|s| s.to_string()))
295        .collect();
296    
297    let decoded_parts = decoded_parts.map_err(|e| TrnError::format(
298        format!("Failed to decode URL components: {}", e),
299        Some(url.to_string()),
300    ))?;
301    
302    // Build TRN string
303    let mut trn_str = format!("trn:{}", decoded_parts.join(":"));
304    if let Some(h) = hash {
305        trn_str.push('@');
306        trn_str.push_str(&h);
307    }
308    
309    parse_trn(&trn_str)
310}

312/// Parse query parameters
313fn parse_query_params(query: &str) -> std::collections::HashMap<String, String> {
314    let mut params = std::collections::HashMap::new();
315    
316    for pair in query.split('&') {
317        if let Some(eq_pos) = pair.find('=') {
318            let key = &pair[..eq_pos];
319            let value = &pair[eq_pos + 1..];
320            
321            // URL decode key and value
322            if let (Ok(decoded_key), Ok(decoded_value)) = (
323                percent_decode_str(key).decode_utf8(),
324                percent_decode_str(value).decode_utf8(),
325            ) {
326                params.insert(decoded_key.to_string(), decoded_value.to_string());
327            }
328        }
329    }
330    
331    params
332}

334/// Check if a component looks like a scope for the given platform
335fn is_scope_like(value: &str, platform: &str) -> bool {
336    match platform {
337        "user" => value.len() >= 2 && value.chars().all(|c| c.is_alphanumeric()),
338        "org" => value.len() >= 2,
339        "aiplatform" => false, // aiplatform typically doesn't have scope
340        _ => value.len() <= 32 && value.chars().all(|c| c.is_alphanumeric() || c == '-'),
341    }
342}
