82// Complete TRN Regex Pattern
83/// Compiled regex for complete TRN validation
84pub static TRN_REGEX: Lazy<Regex> = Lazy::new(|| {
85    let pattern = format!(
86        r"^trn:({platform})(?::({scope}))?:({resource_type}):({type})(?::({subtype}))?:({instance_id}):({version})(?::({tag}))?(?:@({hash}))?$",
87        platform = PLATFORM_PATTERN,
88        scope = SCOPE_PATTERN,
89        resource_type = RESOURCE_TYPE_PATTERN,
90        type = TYPE_PATTERN,
91        subtype = SUBTYPE_PATTERN,
92        instance_id = INSTANCE_ID_PATTERN,
93        version = VERSION_PATTERN,
94        tag = TAG_PATTERN,
95        hash = HASH_PATTERN,
96    );
97    Regex::new(&pattern).unwrap()
98});

105        platform = PLATFORM_PATTERN,
106        scope = SCOPE_PATTERN,
107        resource_type = RESOURCE_TYPE_PATTERN,
108        type = TYPE_PATTERN,
109        subtype = SUBTYPE_PATTERN,

111        version = VERSION_PATTERN,
112        tag = TAG_PATTERN,

114    );

116});

298/// Default configuration values
299#[derive(Debug, Clone)]
300pub struct TrnConfig {
301    /// Whether to validate TRNs on creation
302    pub validate_on_create: bool,
303    /// Whether to use strict validation rules
304    pub strict_validation: bool,
305    /// Whether to allow uppercase characters (converts to lowercase)
306    pub allow_uppercase: bool,
307    /// Whether to normalize TRNs on parsing
308    pub normalize_on_parse: bool,
309    /// Whether to cache validation results
310    pub cache_validation_results: bool,
311    /// Maximum cache size
312    pub max_cache_size: usize,
313    /// Whether to enable deprecation warnings
314    pub enable_deprecation_warnings: bool,
315    /// Whether to verify hashes
316    pub hash_verification: bool,
317    /// Whether to resolve aliases
318    pub alias_resolution: bool,
319    /// Whether to check permissions
320    pub permission_check: bool,
321}

410/// Check if a platform is supported
411pub fn is_platform_supported(platform: &str) -> bool {
412    SUPPORTED_PLATFORMS.contains(platform)
413}
414
415/// Check if a resource type is supported
416pub fn is_resource_type_supported(resource_type: &str) -> bool {
417    SUPPORTED_RESOURCE_TYPES.contains(resource_type)
418}
419
420/// Check if a tool type is supported
421pub fn is_tool_type_supported(tool_type: &str) -> bool {
422    SUPPORTED_TOOL_TYPES.contains(tool_type)
423}
424
425/// Check if a word is reserved
426pub fn is_reserved_word(word: &str) -> bool {
427    RESERVED_WORDS.contains(word) || word.starts_with("__")
428}
429
430/// Check if a version is a common alias
431pub fn is_common_alias(version: &str) -> bool {
432    COMMON_ALIASES.contains(version)
433}
434
435/// Get the expected hash length for an algorithm
436pub fn get_hash_length(algorithm: &str) -> Option<usize> {
437    HASH_ALGORITHMS.get(algorithm).copied()
438}

445    fn test_trn_regex() {
446        let trn = "trn:user:alice:tool:openapi:github-api:v1.0";
